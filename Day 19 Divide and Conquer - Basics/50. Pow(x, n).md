# 50. Pow(x, n)

Difficulty: LeetCode Medium

```java
// This solution is in Java
class Solution {
    public double myPow(double x, int n) {

        if(n < 0){
            n = -n;
            x = 1 / x;
        }

        double pow = 1;

        while(n != 0){
            if((n & 1) != 0){
                pow *= x;
            }

            x *= x;
            n >>>= 1;

        }

        return pow;
    }
}
```
