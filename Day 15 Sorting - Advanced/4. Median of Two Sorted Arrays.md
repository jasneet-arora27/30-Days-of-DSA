# 4. Median of Two Sorted Arrays

Difficulty: LeetCode Hard

```java
// This solution is in Java
class Solution {
    public double findMedianSortedArrays(int[] nums1, int[] nums2) {
        int m = nums1.length;
        int n = nums2.length;

        int[] arr = new int[m + n];

        int i = 0;
        for(int j = 0; j < m ; j++){
            arr[i] = nums1[j];
            i++;  
        }
        for(int j = 0; j < n ; j++){
            arr[i] = nums2[j];
            i++;
        }

        Arrays.sort(arr);

        if ((m + n) % 2 == 0) {
            return(((double) arr[(m + n) / 2] + (double)arr[((m + n) / 2)-1]) / 2);
        }
        else {
            return arr[(m + n - 1) / 2];
        }
        
    }
}
```
